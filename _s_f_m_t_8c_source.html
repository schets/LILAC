167 static void period_certification(sfmt_t * sfmt) {
168  int inner = 0;
169  int i, j;
170  uint32_t work;
171  uint32_t *psfmt32 = &sfmt->state[0].u[0];
172  const uint32_t parity[4] = {SFMT_PARITY1, SFMT_PARITY2,
173  SFMT_PARITY3, SFMT_PARITY4};
174 
175  for (i = 0; i < 4; i++)
176  inner ^= psfmt32[idxof(i)] & parity[i];
177  for (i = 16; i > 0; i >>= 1)
178  inner ^= inner >> i;
179  inner &= 1;
180  /* check OK */
181  if (inner == 1) {
182  return;
183  }
184  /* check NG, and modification */
185  for (i = 0; i < 4; i++) {
186  work = 1;
187  for (j = 0; j < 32; j++) {
188  if ((work & parity[i]) != 0) {
189  psfmt32[idxof(i)] ^= work;
190  return;
191  }
192  work = work << 1;
193  }
194  }
195 }
