00109 def MakeRelativePathsInFlagsAbsolute( flags, working_directory ):
00110   if not working_directory:
00111     return list( flags )
00112   new_flags = []
00113   make_next_absolute = False
00114   path_flags = [ '-isystem', '-I', '-iquote', '--sysroot=' ]
00115   for flag in flags:
00116     new_flag = flag
00117 
00118     if make_next_absolute:
00119       make_next_absolute = False
00120       if not flag.startswith( '/' ):
00121         new_flag = os.path.join( working_directory, flag )
00122 
00123     for path_flag in path_flags:
00124       if flag == path_flag:
00125         make_next_absolute = True
00126         break
00127 
00128       if flag.startswith( path_flag ):
00129         path = flag[ len( path_flag ): ]
00130         new_flag = path_flag + os.path.join( working_directory, path )
00131         break
00132 
00133     if new_flag:
00134       new_flags.append( new_flag )
00135   return new_flags

00143 def GetCompilationInfoForFile( filename ):
00144   # The compilation_commands.json file generated by CMake does not have entries
00145   # for header files. So we do our best by asking the db for flags for a
00146   # corresponding source file, if any. If one exists, the flags for that file
00147   # should be good enough.
00148   if IsHeaderFile( filename ):
00149     basename = os.path.splitext( filename )[ 0 ]
00150     for extension in SOURCE_EXTENSIONS:
00151       replacement_file = basename + extension
00152       if os.path.exists( replacement_file ):
00153         compilation_info = database.GetCompilationInfoForFile(
00154           replacement_file )
00155         if compilation_info.compiler_flags_:
00156           return compilation_info
00157     return None
00158   return database.GetCompilationInfoForFile( filename )
