00120   v4sf invalid_mask = _mm_cmple_ps(x, _mm_setzero_ps());

00215   v4sf tmp = _mm_setzero_ps(), fx;
00216 #ifdef USE_SSE2
00217   v4si emm0;
00218 #else
00219   v2si mm0, mm1;
00220 #endif
00221   v4sf one = *(v4sf*)_ps_1;
00222 
00223   x = _mm_min_ps(x, *(v4sf*)_ps_exp_hi);
00224   x = _mm_max_ps(x, *(v4sf*)_ps_exp_lo);
00225 
00226   /* express exp(x) as exp(g + n*log(2)) */
00227   fx = _mm_mul_ps(x, *(v4sf*)_ps_cephes_LOG2EF);
00228   fx = _mm_add_ps(fx, *(v4sf*)_ps_0p5);
00229 
00230   /* how to perform a floorf with SSE: just below */
00231 #ifndef USE_SSE2
00232   /* step 1 : cast to int */
00233   tmp = _mm_movehl_ps(tmp, fx);
00234   mm0 = _mm_cvttps_pi32(fx);
00235   mm1 = _mm_cvttps_pi32(tmp);
00236   /* step 2 : cast back to float */
00237   tmp = _mm_cvtpi32x2_ps(mm0, mm1);
00238 #else
00239   emm0 = _mm_cvttps_epi32(fx);
00240   tmp  = _mm_cvtepi32_ps(emm0);
00241 #endif
00242   /* if greater, substract 1 */
00243   v4sf mask = _mm_cmpgt_ps(tmp, fx);    
00244   mask = _mm_and_ps(mask, one);
00245   fx = _mm_sub_ps(tmp, mask);
00246 
00247   tmp = _mm_mul_ps(fx, *(v4sf*)_ps_cephes_exp_C1);
00248   v4sf z = _mm_mul_ps(fx, *(v4sf*)_ps_cephes_exp_C2);
00249   x = _mm_sub_ps(x, tmp);
00250   x = _mm_sub_ps(x, z);
00251 
00252   z = _mm_mul_ps(x,x);
00253   
00254   v4sf y = *(v4sf*)_ps_cephes_exp_p0;
00255   y = _mm_mul_ps(y, x);
00256   y = _mm_add_ps(y, *(v4sf*)_ps_cephes_exp_p1);
00257   y = _mm_mul_ps(y, x);
00258   y = _mm_add_ps(y, *(v4sf*)_ps_cephes_exp_p2);
00259   y = _mm_mul_ps(y, x);
00260   y = _mm_add_ps(y, *(v4sf*)_ps_cephes_exp_p3);
00261   y = _mm_mul_ps(y, x);
00262   y = _mm_add_ps(y, *(v4sf*)_ps_cephes_exp_p4);
00263   y = _mm_mul_ps(y, x);
00264   y = _mm_add_ps(y, *(v4sf*)_ps_cephes_exp_p5);
00265   y = _mm_mul_ps(y, z);
00266   y = _mm_add_ps(y, x);
00267   y = _mm_add_ps(y, one);
00268 
00269   /* build 2^n */
00270 #ifndef USE_SSE2

00272   mm0 = _mm_cvttps_pi32(fx);

00304 /* evaluation of 4 sines at onces, using only SSE1+MMX intrinsics so
00305    it runs also on old athlons XPs and the pentium III of your grand
00306    mother.
00307 
00308    The code is the exact rewriting of the cephes sinf function.
00309    Precision is excellent as long as x < 8192 (I did not bother to
00310    take into account the special handling they have for greater values
00311    -- it does not return garbage for arguments over 8192, though, but
00312    the extra precision is missing).
00313 
00314    Note that it is such that sinf((float)M_PI) = 8.74e-8, which is the
00315    surprising but correct result.
00316 
00317    Performance is also surprisingly good, 1.33 times faster than the
00318    macos vsinf SSE2 function, and 1.5 times faster than the
00319    __vrs4_sinf of amd's ACML (which is only available in 64 bits). Not
00320    too bad for an SSE1 function (with no special tuning) !
00321    However the latter libraries probably have a much better handling of NaN,
00322    Inf, denormalized and other special arguments..
00323 
00324    On my core 1 duo, the execution of this function takes approximately 95 cycles.
00325 
00326    From what I have observed on the experiments with Intel AMath lib, switching to an
00327    SSE2 version would improve the perf by only 10%.
00328 
00329    Since it is based on SSE intrinsics, it has to be compiled at -O2 to
00330    deliver full speed.
00331 */
00332 v4sf sin_ps(v4sf x) { // any x
00333   v4sf xmm1, xmm2 = _mm_setzero_ps(), xmm3, sign_bit, y;
00334 
00335 #ifdef USE_SSE2
00336   v4si emm0, emm2;
00337 #else
00338   v2si mm0, mm1, mm2, mm3;
00339 #endif
00340   sign_bit = x;
00341   /* take the absolute value */
00342   x = _mm_and_ps(x, *(v4sf*)_ps_inv_sign_mask);
00343   /* extract the sign bit (upper one) */
00344   sign_bit = _mm_and_ps(sign_bit, *(v4sf*)_ps_sign_mask);
00345   
00346   /* scale by 4/Pi */
00347   y = _mm_mul_ps(x, *(v4sf*)_ps_cephes_FOPI);
00348 
00349 #ifdef USE_SSE2
00350   /* store the integer part of y in mm0 */
00351   emm2 = _mm_cvttps_epi32(y);
00352   /* j=(j+1) & (~1) (see the cephes sources) */
00353   emm2 = _mm_add_epi32(emm2, *(v4si*)_pi32_1);
00354   emm2 = _mm_and_si128(emm2, *(v4si*)_pi32_inv1);
00355   y = _mm_cvtepi32_ps(emm2);
00356 
00357   /* get the swap sign flag */
00358   emm0 = _mm_and_si128(emm2, *(v4si*)_pi32_4);
00359   emm0 = _mm_slli_epi32(emm0, 29);
00360   /* get the polynom selection mask 
00361      there is one polynom for 0 <= x <= Pi/4
00362      and another one for Pi/4<x<=Pi/2
00363 
00364      Both branches will be computed.
00365   */
00366   emm2 = _mm_and_si128(emm2, *(v4si*)_pi32_2);
00367   emm2 = _mm_cmpeq_epi32(emm2, _mm_setzero_si128());
00368   
00369   v4sf swap_sign_bit = _mm_castsi128_ps(emm0);
00370   v4sf poly_mask = _mm_castsi128_ps(emm2);
00371   sign_bit = _mm_xor_ps(sign_bit, swap_sign_bit);
00372   
00373 #else
00374   /* store the integer part of y in mm0:mm1 */
00375   xmm2 = _mm_movehl_ps(xmm2, y);
00376   mm2 = _mm_cvttps_pi32(y);
00377   mm3 = _mm_cvttps_pi32(xmm2);
00378   /* j=(j+1) & (~1) (see the cephes sources) */
00379   mm2 = _mm_add_pi32(mm2, *(v2si*)_pi32_1);
00380   mm3 = _mm_add_pi32(mm3, *(v2si*)_pi32_1);
00381   mm2 = _mm_and_si64(mm2, *(v2si*)_pi32_inv1);
00382   mm3 = _mm_and_si64(mm3, *(v2si*)_pi32_inv1);
00383   y = _mm_cvtpi32x2_ps(mm2, mm3);
00384   /* get the swap sign flag */
00385   mm0 = _mm_and_si64(mm2, *(v2si*)_pi32_4);
00386   mm1 = _mm_and_si64(mm3, *(v2si*)_pi32_4);
00387   mm0 = _mm_slli_pi32(mm0, 29);
00388   mm1 = _mm_slli_pi32(mm1, 29);
00389   /* get the polynom selection mask */
00390   mm2 = _mm_and_si64(mm2, *(v2si*)_pi32_2);
00391   mm3 = _mm_and_si64(mm3, *(v2si*)_pi32_2);
00392   mm2 = _mm_cmpeq_pi32(mm2, _mm_setzero_si64());
00393   mm3 = _mm_cmpeq_pi32(mm3, _mm_setzero_si64());
00394   v4sf swap_sign_bit, poly_mask;
00395   COPY_MM_TO_XMM(mm0, mm1, swap_sign_bit);
00396   COPY_MM_TO_XMM(mm2, mm3, poly_mask);
00397   sign_bit = _mm_xor_ps(sign_bit, swap_sign_bit);
00398   _mm_empty(); /* good-bye mmx */
00399 #endif
00400   
00401   /* The magic pass: "Extended precision modular arithmetic" 
00402      x = ((x - y * DP1) - y * DP2) - y * DP3; */
00403   xmm1 = *(v4sf*)_ps_minus_cephes_DP1;
00404   xmm2 = *(v4sf*)_ps_minus_cephes_DP2;
00405   xmm3 = *(v4sf*)_ps_minus_cephes_DP3;
00406   xmm1 = _mm_mul_ps(y, xmm1);
00407   xmm2 = _mm_mul_ps(y, xmm2);
00408   xmm3 = _mm_mul_ps(y, xmm3);
00409   x = _mm_add_ps(x, xmm1);
00410   x = _mm_add_ps(x, xmm2);
00411   x = _mm_add_ps(x, xmm3);
00412 
00413   /* Evaluate the first polynom  (0 <= x <= Pi/4) */
00414   y = *(v4sf*)_ps_coscof_p0;
00415   v4sf z = _mm_mul_ps(x,x);
00416 
00417   y = _mm_mul_ps(y, z);
00418   y = _mm_add_ps(y, *(v4sf*)_ps_coscof_p1);
00419   y = _mm_mul_ps(y, z);
00420   y = _mm_add_ps(y, *(v4sf*)_ps_coscof_p2);
00421   y = _mm_mul_ps(y, z);
00422   y = _mm_mul_ps(y, z);
00423   v4sf tmp = _mm_mul_ps(z, *(v4sf*)_ps_0p5);
00424   y = _mm_sub_ps(y, tmp);
00425   y = _mm_add_ps(y, *(v4sf*)_ps_1);
00426   
00427   /* Evaluate the second polynom  (Pi/4 <= x <= 0) */
00428 
00429   v4sf y2 = *(v4sf*)_ps_sincof_p0;
00430   y2 = _mm_mul_ps(y2, z);
00431   y2 = _mm_add_ps(y2, *(v4sf*)_ps_sincof_p1);
00432   y2 = _mm_mul_ps(y2, z);
00433   y2 = _mm_add_ps(y2, *(v4sf*)_ps_sincof_p2);
00434   y2 = _mm_mul_ps(y2, z);
00435   y2 = _mm_mul_ps(y2, x);
00436   y2 = _mm_add_ps(y2, x);
00437 
00438   /* select the correct result from the two polynoms */  
00439   xmm3 = poly_mask;
00440   y2 = _mm_and_ps(xmm3, y2); //, xmm3);
00441   y = _mm_andnot_ps(xmm3, y);
00442   y = _mm_add_ps(y,y2);
00443   /* update the sign */
00444   y = _mm_xor_ps(y, sign_bit);
00445   return y;
00446 }
