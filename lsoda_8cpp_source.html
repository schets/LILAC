00210    is defined in a similar way using incy.
00211 
00212 */

00215     double          dotprod;
00216     int             ix, iy, i, m;
00217 
00218     dotprod = 0.;

00220         return dotprod;
00221 
00222 /* Code for unequal or nonpositive increments.  */
00223

00225         ix = 1;

00228             ix = (-n + 1) * incx + 1;

00230             iy = (-n + 1) * incy + 1;

00364 /*
00365    Purpose : dgesl solves the linear system
00366    a * x = b or Transpose(a) * x = b
00367    using the factors computed by dgeco or degfa.
00368 
00369 
00370    On Entry :
00371 
00372       a    : double matrix of dimension ( n+1, n+1 ),
00373              the output from dgeco or dgefa.
00374              The 0-th row and column are not used.
00375       n    : the row dimension of a.
00376       ipvt : the pivot vector from degco or dgefa.
00377       b    : the right hand side vector.
00378       job  : = 0       to solve a * x = b,
00379              = nonzero to solve Transpose(a) * x = b.
00380 
00381 
00382    On Return :
00383 
00384       b : the solution vector x.
00385 
00386 
00387    Error Condition :
00388 
00389       A division by zero will occur if the input factor contains
00390       a zero on the diagonal.  Technically this indicates
00391       singularity but it is often caused by improper argments or
00392       improper setting of the pointers of a.  It will not occur
00393       if the subroutines are called correctly and if dgeco has
00394       set rcond > 0 or dgefa has set info = 0.
00395 
00396 
00397    BLAS : daxpy, ddot
00398 */
00399 
00400 {
00401     int             nm1, k, j;
00402     double          t;
00403 
00404     nm1 = n - 1;
00405 
00406 /*
00407    Job = 0, solve a * x = b.
00408 */
00409     if (job == 0) {
00410 /*
00411    First solve L * y = b.
00412 */
00413         for (k = 1; k <= n; k++) {
00414             t = ddot(k - 1, a[k], 1, b, 1);
00415             b[k] = (b[k] - t) / a[k][k];
00416         }
00417 /*
00418    Now solve U * x = y.
00419 */
00420         for (k = n - 1; k >= 1; k--) {
00421             b[k] = b[k] + ddot(n - k, a[k] + k, 1, b + k, 1);
00422             j = ipvt[k];
00423             if (j != k) {
00424                 t = b[j];
00425                 b[j] = b[k];
00426                 b[k] = t;
00427             }
00428         }
00429         return;
00430     }
00431 /*
00432    Job = nonzero, solve Transpose(a) * x = b.
00433 
00434    First solve Transpose(U) * y = b.
00435 */
00436     for (k = 1; k <= n - 1; k++) {
00437         j = ipvt[k];
00438         t = b[j];
00439         if (j != k) {
00440             b[j] = b[k];
00441             b[k] = t;
00442         }
00443         daxpy(n - k, t, a[k] + k, 1, b + k, 1);
00444     }
00445 /*
00446    Now solve Transpose(L) * x = y.
00447 */
00448     for (k = n; k >= 1; k--) {
00449         b[k] = b[k] / a[k][k];
00450         t = -b[k];

00452     }

00454 }

00461 dgefa(double** a, int n, int* ipvt, int* info)
00462 
00463 /*
00464    Purpose : dgefa factors a double matrix by Gaussian elimination.
00465 
00466    dgefa is usually called by dgeco, but it can be called directly
00467    with a saving in time if rcond is not needed.
00468    (Time for dgeco) = (1+9/n)*(time for dgefa).
00469 
00470    This c version uses algorithm kji rather than the kij in dgefa.f.
00471    Note that the fortran version input variable lda is not needed.
00472 
00473 
00474    On Entry :
00475 
00476       a   : double matrix of dimension ( n+1, n+1 ),
00477             the 0-th row and column are not used.
00478             a is created using NewDoubleMatrix, hence
00479             lda is unnecessary.
00480       n   : the row dimension of a.
00481 
00482    On Return :
00483 
00484       a     : a lower triangular matrix and the multipliers
00485               which were used to obtain it.  The factorization
00486               can be written a = L * U where U is a product of
00487               permutation and unit upper triangular matrices
00488               and L is lower triangular.
00489       ipvt  : an n+1 integer vector of pivot indices.
00490       *info : = 0 normal value,
00491               = k if U[k][k] == 0.  This is not an error
00492                 condition for this subroutine, but it does
00493                 indicate that dgesl or dgedi will divide by
00494                 zero if called.  Use rcond in dgeco for
00495                 a reliable indication of singularity.
00496 
00497                 Notice that the calling program must use &info.
00498 
00499    BLAS : daxpy, dscal, idamax
00500 */
00501 
00502 {
00503     int             j, k, i;
00504     double          t;
00505 
00506 /* Gaussian elimination with partial pivoting.   */
00507 
00508     *info = 0;
00509     for (k = 1; k <= n - 1; k++) {
00510 /*
00511    Find j = pivot index.  Note that a[k]+k-1 is the address of
00512    the 0-th element of the row vector whose 1st element is a[k][k].
00513 */
00514         j = idamax(n - k + 1, a[k] + k - 1, 1) + k - 1;
00515         ipvt[k] = j;
00516 /*
00517    Zero pivot implies this row already triangularized.
00518 */
00519         if (a[k][j] == 0.) {
00520             *info = k;
00521             continue;
00522         }
00523 /*
00524    Interchange if necessary.
00525 */
00526         if (j != k) {
00527             t = a[k][j];
00528             a[k][j] = a[k][k];
00529             a[k][k] = t;
00530         }
00531 /*
00532    Compute multipliers.
00533 */
00534         t = -1. / a[k][k];

00536 /*
00537    Column elimination with row indexing.
00538 */
00539         for (i = k + 1; i <= n; i++) {
00540             t = a[i][j];
00541             if (j != k) {
00542                 a[i][j] = a[i][k];
00543                 a[i][k] = t;
00544             }
00545             daxpy(n - k, t, a[k] + k, 1, a[i] + k, 1);
00546         }
00547     }           /* end k-loop  */
00548 
00549     ipvt[n] = n;
00550     if (a[n][n] == 0.)
00551         *info = n;
00552 
00553 }

01294                 return;
01295             }
01296             break;
01297         case 2:

01301             if ((tp - tout) * h > 0.) {
01302                 fprintf(stderr, "[lsoda] itask = %d and tout behind tcur - hu\n", itask);
01303                 terminate(istate);
01304                 freevectors();
01305                 return;
01306             }
01307             if ((tn - tout) * h < 0.) break;
01308             successreturn(y, t, itask, ihit, tcrit, istate);
01309             return;
01310         case 4:
01311             tcrit = rwork1;
01312             if ((tn - tcrit) * h > 0.) {
01313                 fprintf(stderr, "[lsoda] itask = 4 or 5 and tcrit behind tcur\n");
01314                 terminate(istate);
01315                 freevectors();
01316                 return;
01317             }
01318             if ((tcrit - tout) * h < 0.) {
01319                 fprintf(stderr, "[lsoda] itask = 4 or 5 and tcrit behind tout\n");
01320                 terminate(istate);
01321                 freevectors();
01322                 return;
01323             }
01324             if ((tn - tout) * h >= 0.) {
01325                 intdy(tout, 0, y, &iflag);
01326                 if (iflag != 0) {
01327                     fprintf(stderr, "[lsoda] trouble from intdy, itask = %d, tout = %g\n", itask, tout);
01328                     terminate(istate);
01329                     freevectors();
01330                     return;
01331                 }
01332                 *t = tout;
01333                 *istate = 2;
01334                 illin = 0;
01335                 freevectors();
01336                 return;
01337             }
01338         case 5:
01339             if (itask == 5) {
01340                 tcrit = rwork1;
01341                 if ((tn - tcrit) * h > 0.) {
01342                     fprintf(stderr, "[lsoda] itask = 4 or 5 and tcrit behind tcur\n");
01343                     terminate(istate);
01344                     freevectors();
01345                     return;
01346                 }
01347             }
01348             hmx = fabs(tn) + fabs(h);
01349             ihit = fabs(tn - tcrit) <= (100. * ETA * hmx);
01350             if (ihit) {
01351                 *t = tcrit;
01352                 successreturn(y, t, itask, ihit, tcrit, istate);
01353                 return;
01354             }
01355             tnext = tn + h * (1. + 4. * ETA);
01356             if ((tnext - tcrit) * h <= 0.)
01357                 break;

01359             if (*istate == 2)
01360                 jstart = -2;
01361             break;
01362         }       /* end switch   */
01363     }           /* end if ( *istate == 2 || *istate == 3 )   */
01364     /*
01365        Block e.
01366        The next block is normally executed for all calls and contains
01367        the call to the one-step core integrator stoda.
01368     
01369        This is a looping point for the integration steps.
01370     
01371        First check for too many steps being taken, update ewt ( if not at
01372        start of problem).  Check for too much accuracy being requested, and
01373        check for h below the roundoff level in *t.
01374     */
01375     while (1) {

01377             if ((nst - nslast) >= mxstep) {
01378                 fprintf(stderr, "[lsoda] %d steps taken before reaching tout\n", mxstep);
01379                 *istate = -1;
01380                 terminate2(y, t);
01381                 return;
01382             }
01383             ewset(itol, rtol, atol, yh[1]);

01385                 if (ewt[i] <= 0.) {
01386                     fprintf(stderr, "[lsoda] ewt[%d] = %g <= 0.\n", i, ewt[i]);
01387                     *istate = -6;
01388                     terminate2(y, t);
01389                     return;
01390                 }
01391                 ewt[i] = 1. / ewt[i];
01392             }
01393         }

01398                 fprintf(stderr, "lsoda -- at start of problem, too much accuracy\n");
01399                 fprintf(stderr, "         requested for precision of machine,\n");
01400                 fprintf(stderr, "         suggested scaling factor = %g\n", tolsf);
01401                 terminate(istate);
01402                 freevectors();
01403                 return;
01404             }
01405             fprintf(stderr, "lsoda -- at t = %g, too much accuracy requested\n", *t);
01406             fprintf(stderr, "         for precision of machine, suggested\n");
01407             fprintf(stderr, "         scaling factor = %g\n", tolsf);
01408             *istate = -2;
01409             terminate2(y, t);
01410             return;
01411         }
01412         if ((tn + h) == tn) {
01413             nhnil++;
01414             if (nhnil <= mxhnil) {
01415                 fprintf(stderr, "lsoda -- warning..internal t = %g and h = %g are\n", tn, h);
01416                 fprintf(stderr, "         such that in the machine, t + h = t on the next step\n");
01417                 fprintf(stderr, "         solver will continue anyway.\n");
01418                 if (nhnil == mxhnil) {
01419                     fprintf(stderr, "lsoda -- above warning has been issued %d times,\n", nhnil);
01420                     fprintf(stderr, "         it will not be issued again for this problem\n");
01421                 }
01422             }
01423         }
01424 /*
01425    Call stoda
01426 */
01427         stoda(neq, y, f, _data);
01428 
01429 /*
01430    printf( "meth= %d,   order= %d,   nfe= %d,   nje= %d\n",
01431       meth, nq, nfe, nje );
01432    printf( "t= %20.15e,   h= %20.15e,   nst=%d\n", tn, h, nst );
01433    printf( "y= %20.15e,   %20.15e,   %20.15e\n\n\n",
01434       yh[1][1], yh[1][2], yh[1][3] );
01435 */
01436 
01437         if (kflag == 0) {
01438 /*
01439    Block f.
01440    The following block handles the case of a successful return from the
01441    core integrator ( kflag = 0 ).
01442    If a method switch was just made, record tsw, reset maxord,
01443    set jstart to -1 to signal stoda to complete the switch,
01444    and do extra printing of data if ixpr = 1.
01445    Then, in any case, check for stop conditions.
01446 */
01447             init = 1;
01448             if (meth != mused) {
01449                 tsw = tn;
01450                 maxord = mxordn;
01451                 if (meth == 2)
01452                     maxord = mxords;
01453                 jstart = -1;
01454                 if (ixpr) {
01455                     if (meth == 2)
01456                         fprintf(stderr, "[lsoda] a switch to the stiff method has occurred ");
01457                     if (meth == 1)
01458                         fprintf(stderr, "[lsoda] a switch to the nonstiff method has occurred");
01459                     fprintf(stderr, "at t = %g, tentative step size h = %g, step nst = %d\n", tn, h, nst);
01460                 }
01461             }   /* end if ( meth != mused )   */
01462             /*
01463                itask = 1.
01464                If tout has been reached, interpolate.
01465             */
01466             if (itask == 1) {
01467                 if ((tn - tout) * h < 0.)
01468                     continue;
01469                 intdy(tout, 0, y, &iflag);
01470                 *t = tout;
01471                 *istate = 2;
01472                 illin = 0;
01473                 freevectors();
01474                 return;
01475             }
01476 /*
01477    itask = 2.
01478 */
01479             if (itask == 2) {
01480                 successreturn(y, t, itask, ihit, tcrit, istate);
01481                 return;
01482             }
01483 /*
01484    itask = 3.
01485    Jump to exit if tout was reached.
01486 */
01487             if (itask == 3) {
01488                 if ((tn - tout) * h >= 0.) {
01489                     successreturn(y, t, itask, ihit, tcrit, istate);
01490                     return;
01491                 }
01492                 continue;
01493             }
01494 /*
01495    itask = 4.
01496    See if tout or tcrit was reached.  Adjust h if necessary.
01497 */
01498             if (itask == 4) {
01499                 if ((tn - tout) * h >= 0.) {
01500                     intdy(tout, 0, y, &iflag);
01501                     *t = tout;
01502                     *istate = 2;
01503                     illin = 0;
01504                     freevectors();
01505                     return;
01506                 } else {
01507                     hmx = fabs(tn) + fabs(h);
01508                     ihit = fabs(tn - tcrit) <= (100. * ETA * hmx);
01509                     if (ihit) {
01510                         successreturn(y, t, itask, ihit, tcrit, istate);
01511                         return;
01512                     }
01513                     tnext = tn + h * (1. + 4. * ETA);
01514                     if ((tnext - tcrit) * h <= 0.)
01515                         continue;
01516                     h = (tcrit - tn) * (1. - 4. * ETA);
01517                     jstart = -2;
01518                     continue;
01519                 }
01520             }   /* end if ( itask == 4 )   */
01521             /*
01522                itask = 5.
01523                See if tcrit was reached and jump to exit.
01524             */
01525             if (itask == 5) {
01526                 hmx = fabs(tn) + fabs(h);
01527                 ihit = fabs(tn - tcrit) <= (100. * ETA * hmx);
01528                 successreturn(y, t, itask, ihit, tcrit, istate);
01529                 return;
01530             }
01531         }       /* end if ( kflag == 0 )   */
01532         /*
01533            kflag = -1, error test failed repeatedly or with fabs(h) = hmin.
01534            kflag = -2, convergence failed repeatedly or with fabs(h) = hmin.
01535         */
01536         if (kflag == -1 || kflag == -2) {
01537             fprintf(stderr, "lsoda -- at t = %g and step size h = %g, the\n", tn, h);
01538             if (kflag == -1) {
01539                 fprintf(stderr, "         error test failed repeatedly or\n");
01540                 fprintf(stderr, "         with fabs(h) = hmin\n");
01541                 *istate = -4;
01542             }
01543             if (kflag == -2) {
01544                 fprintf(stderr, "         corrector convergence failed repeatedly or\n");
01545                 fprintf(stderr, "         with fabs(h) = hmin\n");
01546                 *istate = -5;
01547             }
01548             big = 0.;
01549             imxer = 1;
01550             for (i = 1; i <= n; i++) {
01551                 size = fabs(acor[i]) * ewt[i];
01552                 if (big < size) {
01553                     big = size;
01554                     imxer = i;
01555                 }
01556             }
01557             terminate2(y, t);
01558             return;
01559         }       /* end if ( kflag == -1 || kflag == -2 )   */
01560     }           /* end while   */
01561 
01562 }               /* end lsoda   */

01565 static void stoda(int neq, double *y, _lsoda_f f, void *_data)
01566 {
01567     int             corflag, orderflag;
01568     int             i, i1, j, m, ncf;
01569     double          del, delp, dsm, dup, exup, r, rh, rhup, told;
01570     double          pdh, pnorm;
01571 
01572 /*
01573    stoda performs one step of the integration of an initial value
01574    problem for a system of ordinary differential equations.
01575    Note.. stoda is independent of the value of the iteration method
01576    indicator miter, when this is != 0, and hence is independent
01577    of the type of chord method used, or the Jacobian structure.
01578    Communication with stoda is done with the following variables:
01579 
01580    jstart = an integer used for input only, with the following
01581             values and meanings:
01582 
01583                0  perform the first step,
01584              > 0  take a new step continuing from the last,
01585               -1  take the next step with a new value of h,
01586                   n, meth, miter, and/or matrix parameters.
01587               -2  take the next step with a new value of h,
01588                   but with other inputs unchanged.
01589 
01590    kflag = a completion code with the following meanings:
01591 
01592              0  the step was successful,
01593             -1  the requested error could not be achieved,
01594             -2  corrector convergence could not be achieved,
01595             -3  fatal error in prja or solsy.
01596 
01597    miter = corrector iteration method:
01598 
01599              0  functional iteration,
01600             >0  a chord method corresponding to jacobian type jt.
01601 
01602 */
01603     kflag = 0;
01604     told = tn;
01605     ncf = 0;
01606     ierpj = 0;
01607     iersl = 0;
01608     jcur = 0;
01609     delp = 0.;
01610 
01611 /*
01612    On the first call, the order is set to 1, and other variables are
01613    initialized.  rmax is the maximum ratio by which h can be increased
01614    in a single step.  It is initially 1.e4 to compensate for the small
01615    initial h, but then is normally equal to 10.  If a filure occurs
01616    (in corrector convergence or error test), rmax is set at 2 for
01617    the next increase.
01618    cfode is called to get the needed coefficients for both methods.
01619 */
01620     if (jstart == 0) {
01621         lmax = maxord + 1;
01622         nq = 1;
01623         l = 2;
01624         ialth = 2;
01625         rmax = 10000.;
01626         rc = 0.;
01627         el0 = 1.;
01628         crate = 0.7;
01629         hold = h;
01630         nslp = 0;
01631         ipup = miter;
01632 /*
01633    Initialize switching parameters.  meth = 1 is assumed initially.
01634 */
01635         icount = 20;
01636         irflag = 0;
01637         pdest = 0.;
01638         pdlast = 0.;
01639         ratio = 5.;
01640         cfode(2);
01641         for (i = 1; i <= 5; i++)
01642             cm2[i] = tesco[i][2] * elco[i][i + 1];
01643         cfode(1);
01644         for (i = 1; i <= 12; i++)
01645             cm1[i] = tesco[i][2] * elco[i][i + 1];
01646         resetcoeff();
01647     }           /* end if ( jstart == 0 )   */
01648     /*
01649        The following block handles preliminaries needed when jstart = -1.
01650        ipup is set to miter to force a matrix update.
01651        If an order increase is about to be considered ( ialth = 1 ),
01652        ialth is reset to 2 to postpone consideration one more step.
01653        If the caller has changed meth, cfode is called to reset
01654        the coefficients of the method.
01655        If h is to be changed, yh must be rescaled.
01656        If h or meth is being changed, ialth is reset to l = nq + 1
01657        to prevent further changes in h for that many steps.
01658     */
01659     if (jstart == -1) {
01660         ipup = miter;
01661         lmax = maxord + 1;
01662         if (ialth == 1)
01663             ialth = 2;
01664         if (meth != mused) {
01665             cfode(meth);
01666             ialth = l;
01667             resetcoeff();
01668         }
01669         if (h != hold) {
01670             rh = h / hold;
01671             h = hold;
01672             scaleh(&rh, &pdh);
01673         }
01674     }           /* if ( jstart == -1 )   */
01675     if (jstart == -2) {
01676         if (h != hold) {
01677             rh = h / hold;
01678             h = hold;
01679             scaleh(&rh, &pdh);
01680         }
01681     }           /* if ( jstart == -2 )   */
01682     /*
01683        Prediction.
01684        This section computes the predicted values by effectively
01685        multiplying the yh array by the pascal triangle matrix.
01686        rc is the ratio of new to old values of the coefficient h * el[1].
01687        When rc differs from 1 by more than ccmax, ipup is set to miter
01688        to force pjac to be called, if a jacobian is involved.
01689        In any case, prja is called at least every msbp steps.
01690     */
01691     while (1) {
01692         while (1) {
01693             if (fabs(rc - 1.) > ccmax)
01694                 ipup = miter;
01695             if (nst >= nslp + msbp)
01696                 ipup = miter;
01697             tn += h;
01698             for (j = nq; j >= 1; j--)
01699                 for (i1 = j; i1 <= nq; i1++) {
01700                     yp1 = yh[i1];
01701                     yp2 = yh[i1 + 1];
01702                     for (i = 1; i <= n; i++)
01703                         yp1[i] += yp2[i];
01704                 }
01705             pnorm = vmnorm(n, yh[1], ewt);
01706 
01707             correction(neq, y, f, &corflag, pnorm, &del, &delp, &told, &ncf, &rh, &m, _data);
01708             if (corflag == 0)
01709                 break;
01710             if (corflag == 1) {
01711                 rh = max(rh, hmin / fabs(h));
01712                 scaleh(&rh, &pdh);
01713                 continue;
01714             }
01715             if (corflag == 2) {
01716                 kflag = -2;
01717                 hold = h;
01718                 jstart = 1;
01719                 return;
01720             }
01721         }       /* end inner while ( corrector loop )   */
01722 /*
01723    The corrector has converged.  jcur is set to 0
01724    to signal that the Jacobian involved may need updating later.
01725    The local error test is done now.
01726 */
01727         jcur = 0;
01728         if (m == 0)
01729             dsm = del / tesco[nq][2];
01730         if (m > 0)
01731             dsm = vmnorm(n, acor, ewt) / tesco[nq][2];
01732         if (dsm <= 1.) {
01733 /*
01734    After a successful step, update the yh array.
01735    Decrease icount by 1, and if it is -1, consider switching methods.
01736    If a method switch is made, reset various parameters,
01737    rescale the yh array, and exit.  If there is no switch,
01738    consider changing h if ialth = 1.  Otherwise decrease ialth by 1.
01739    If ialth is then 1 and nq < maxord, then acor is saved for
01740    use in a possible order increase on the next step.
01741    If a change in h is considered, an increase or decrease in order
01742    by one is considered also.  A change in h is made only if it is by
01743    a factor of at least 1.1.  If not, ialth is set to 3 to prevent
01744    testing for that many steps.
01745 */
01746             kflag = 0;
01747             nst++;
01748             hu = h;
01749             nqu = nq;
01750             mused = meth;
01751             for (j = 1; j <= l; j++) {
01752                 yp1 = yh[j];
01753                 r = el[j];
01754                 for (i = 1; i <= n; i++)
01755                     yp1[i] += r * acor[i];
01756             }
01757             icount--;
01758             if (icount < 0) {
01759                 methodswitch(dsm, pnorm, &pdh, &rh);
01760                 if (meth != mused) {
01761                     rh = max(rh, hmin / fabs(h));
01762                     scaleh(&rh, &pdh);
01763                     rmax = 10.;
01764                     endstoda();
01765                     break;
01766                 }
01767             }
01768 /*
01769    No method switch is being made.  Do the usual step/order selection.
01770 */
01771             ialth--;
01772             if (ialth == 0) {
01773                 rhup = 0.;
01774                 if (l != lmax) {
01775                     yp1 = yh[lmax];
01776                     for (i = 1; i <= n; i++)
01777                         savf[i] = acor[i] - yp1[i];
01778                     dup = vmnorm(n, savf, ewt) / tesco[nq][3];
01779                     exup = 1. / (double) (l + 1);
01780                     rhup = 1. / (1.4 * pow(dup, exup) + 0.0000014);
01781                 }
01782                 orderswitch(&rhup, dsm, &pdh, &rh, &orderflag);
01783 /*
01784    No change in h or nq.
01785 */
01786                 if (orderflag == 0) {
01787                     endstoda();
01788                     break;
01789                 }
01790 /*
01791    h is changed, but not nq.
01792 */
01793                 if (orderflag == 1) {
01794                     rh = max(rh, hmin / fabs(h));
01795                     scaleh(&rh, &pdh);
01796                     rmax = 10.;
01797                     endstoda();
01798                     break;
01799                 }
01800 /*
01801    both nq and h are changed.
01802 */
01803                 if (orderflag == 2) {
01804                     resetcoeff();
01805                     rh = max(rh, hmin / fabs(h));
01806                     scaleh(&rh, &pdh);
01807                     rmax = 10.;
01808                     endstoda();
01809                     break;
01810                 }
01811             }   /* end if ( ialth == 0 )   */
01812             if (ialth > 1 || l == lmax) {
01813                 endstoda();
01814                 break;
01815             }
01816             yp1 = yh[lmax];
01817             for (i = 1; i <= n; i++)
01818                 yp1[i] = acor[i];
01819             endstoda();
01820             break;
01821         }
01822         /* end if ( dsm <= 1. )   */
01823         /*
01824            The error test failed.  kflag keeps track of multiple failures.
01825            Restore tn and the yh array to their previous values, and prepare
01826            to try the step again.  Compute the optimum step size for this or
01827            one lower.  After 2 or more failures, h is forced to decrease
01828            by a factor of 0.2 or less.
01829          */ 
01830         else {
01831             kflag--;
01832             tn = told;
01833             for (j = nq; j >= 1; j--)
01834                 for (i1 = j; i1 <= nq; i1++) {
01835                     yp1 = yh[i1];
01836                     yp2 = yh[i1 + 1];
01837                     for (i = 1; i <= n; i++)
01838                         yp1[i] -= yp2[i];
01839                 }
01840             rmax = 2.;
01841             if (fabs(h) <= hmin * 1.00001) {
01842                 kflag = -1;
01843                 hold = h;
01844                 jstart = 1;
01845                 break;
01846             }
01847             if (kflag > -3) {
01848                 rhup = 0.;
01849                 orderswitch(&rhup, dsm, &pdh, &rh, &orderflag);
01850                 if (orderflag == 1 || orderflag == 0) {
01851                     if (orderflag == 0)
01852                         rh = min(rh, 0.2);
01853                     rh = max(rh, hmin / fabs(h));
01854                     scaleh(&rh, &pdh);
01855                 }
01856                 if (orderflag == 2) {
01857                     resetcoeff();
01858                     rh = max(rh, hmin / fabs(h));
01859                     scaleh(&rh, &pdh);
01860                 }
01861                 continue;
01862             }
01863             /* if ( kflag > -3 )   */
01864             /*
01865                Control reaches this section if 3 or more failures have occurred.
01866                If 10 failures have occurred, exit with kflag = -1.
01867                It is assumed that the derivatives that have accumulated in the
01868                yh array have errors of the wrong order.  Hence the first
01869                derivative is recomputed, and the order is set to 1.  Then
01870                h is reduced by a factor of 10, and the step is retried,
01871                until it succeeds or h reaches hmin.
01872              */ 
01873             else {
01874                 if (kflag == -10) {

01931 static void intdy(double t, int k, double *dky, int *iflag)
01932 
01933 /*
01934    Intdy computes interpolated values of the k-th derivative of the
01935    dependent variable vector y, and stores it in dky.  This routine
01936    is called within the package with k = 0 and *t = tout, but may
01937    also be called by the user for any k up to the current order.
01938    ( See detailed instructions in the usage documentation. )
01939 
01940    The computed values in dky are gotten by interpolation using the
01941    Nordsieck history array yh.  This array corresponds uniquely to a
01942    vector-valued polynomial of degree nqcur or less, and dky is set
01943    to the k-th derivative of this polynomial at t.
01944    The formula for dky is
01945 
01946              q
01947    dky[i] = sum c[k][j] * ( t - tn )^(j-k) * h^(-j) * yh[j+1][i]
01948             j=k
01949 
01950    where c[k][j] = j*(j-1)*...*(j-k+1), q = nqcur, tn = tcur, h = hcur.
01951    The quantities nq = nqcur, l = nq+1, n = neq, tn, and h are declared
01952    static globally.  The above sum is done in reverse order.
01953    *iflag is returned negative if either k or t is out of bounds.
01954 */
01955 
01956 {
01957     int             i, ic, j, jj, jp1;
01958     double          c, r, s, tp;
01959 
01960     *iflag = 0;
01961     if (k < 0 || k > nq) {
01962         fprintf(stderr, "[intdy] k = %d illegal\n", k);
01963         *iflag = -1;
01964         return;
01965     }
01966     tp = tn - hu - 100. * ETA * (tn + hu);
01967     if ((t - tp) * (t - tn) > 0.) {
01968         fprintf(stderr, "intdy -- t = %g illegal. t not in interval tcur - hu to tcur\n", t);
01969         *iflag = -2;
01970         return;
01971     }
01972     s = (t - tn) / h;
01973     ic = 1;
01974     for (jj = l - k; jj <= nq; jj++)
01975         ic *= jj;
01976     c = (double) ic;
01977     yp1 = yh[l];
01978     for (i = 1; i <= n; i++)
01979         dky[i] = c * yp1[i];
01980     for (j = nq - 1; j >= k; j--) {
01981         jp1 = j + 1;
01982         ic = 1;
01983         for (jj = jp1 - k; jj <= j; jj++)
01984             ic *= jj;
01985         c = (double) ic;
01986         yp1 = yh[jp1];
01987         for (i = 1; i <= n; i++)
01988             dky[i] = c * yp1[i] + s * dky[i];
01989     }
01990     if (k == 0)
01991         return;
01992     r = pow(h, (double) (-k));
01993     for (i = 1; i <= n; i++)
01994         dky[i] *= r;
01995 
01996 }               /* end intdy   */

01998 static void cfode(int meth)
01999 {
02000     int             i, nq, nqm1, nqp1;
02001     double          agamq, fnq, fnqm1, pc[13], pint, ragq, rqfac, rq1fac, tsign, xpin;
02002 /*
02003    cfode is called by the integrator routine to set coefficients
02004    needed there.  The coefficients for the current method, as
02005    given by the value of meth, are set for all orders and saved.
02006    The maximum order assumed here is 12 if meth = 1 and 5 if meth = 2.
02007    ( A smaller value of the maximum order is also allowed. )
02008    cfode is called once at the beginning of the problem, and
02009    is not called again unless and until meth is changed.
02010 
02011    The elco array contains the basic method coefficients.
02012    The coefficients el[i], 1 < i < nq+1, for the method of
02013    order nq are stored in elco[nq][i].  They are given by a generating
02014    polynomial, i.e.,
02015 
02016       l(x) = el[1] + el[2]*x + ... + el[nq+1]*x^nq.
02017 
02018    For the implicit Adams method, l(x) is given by
02019 
02020       dl/dx = (x+1)*(x+2)*...*(x+nq-1)/factorial(nq-1),   l(-1) = 0.
02021 
02022    For the bdf methods, l(x) is given by
02023 
02024       l(x) = (x+1)*(x+2)*...*(x+nq)/k,
02025 
02026    where   k = factorial(nq)*(1+1/2+...+1/nq).
02027 
02028    The tesco array contains test constants used for the
02029    local error test and the selection of step size and/or order.
02030    At order nq, tesco[nq][k] is used for the selection of step
02031    size at order nq-1 if k = 1, at order nq if k = 2, and at order
02032    nq+1 if k = 3.
02033 */
02034     if (meth == 1) {
02035         elco[1][1] = 1.;
02036         elco[1][2] = 1.;
02037         tesco[1][1] = 0.;
02038         tesco[1][2] = 2.;
02039         tesco[2][1] = 1.;
02040         tesco[12][3] = 0.;
02041         pc[1] = 1.;
02042         rqfac = 1.;
02043         for (nq = 2; nq <= 12; nq++) {
02044 /*
02045    The pc array will contain the coefficients of the polynomial
02046 
02047       p(x) = (x+1)*(x+2)*...*(x+nq-1).
02048 
02049    Initially, p(x) = 1.
02050 */
02051             rq1fac = rqfac;
02052             rqfac = rqfac / (double) nq;
02053             nqm1 = nq - 1;
02054             fnqm1 = (double) nqm1;
02055             nqp1 = nq + 1;
02056 /*
02057    Form coefficients of p(x)*(x+nq-1).
02058 */
02059             pc[nq] = 0.;
02060             for (i = nq; i >= 2; i--)
02061                 pc[i] = pc[i - 1] + fnqm1 * pc[i];
02062             pc[1] = fnqm1 * pc[1];
02063 /*
02064    Compute integral, -1 to 0, of p(x) and x*p(x).
02065 */
02066             pint = pc[1];
02067             xpin = pc[1] / 2.;
02068             tsign = 1.;
02069             for (i = 2; i <= nq; i++) {
02070                 tsign = -tsign;
02071                 pint += tsign * pc[i] / (double) i;
02072                 xpin += tsign * pc[i] / (double) (i + 1);
02073             }
02074 /*
02075    Store coefficients in elco and tesco.
02076 */
02077             elco[nq][1] = pint * rq1fac;
02078             elco[nq][2] = 1.;
02079             for (i = 2; i <= nq; i++)
02080                 elco[nq][i + 1] = rq1fac * pc[i] / (double) i;
02081             agamq = rqfac * xpin;
02082             ragq = 1. / agamq;
02083             tesco[nq][2] = ragq;
02084             if (nq < 12)
02085                 tesco[nqp1][1] = ragq * rqfac / (double) nqp1;
02086             tesco[nqm1][3] = ragq;
02087         }       /* end for   */
02088         return;
02089     }           /* end if ( meth == 1 )   */
02090     /*
02091        meth = 2.
02092     */
02093     pc[1] = 1.;
02094     rq1fac = 1.;
02095 /*
02096    The pc array will contain the coefficients of the polynomial
02097 
02098       p(x) = (x+1)*(x+2)*...*(x+nq).
02099 
02100    Initially, p(x) = 1.
02101 */
02102     for (nq = 1; nq <= 5; nq++) {
02103         fnq = (double) nq;
02104         nqp1 = nq + 1;
02105 /*
02106    Form coefficients of p(x)*(x+nq).
02107 */
02108         pc[nqp1] = 0.;
02109         for (i = nq + 1; i >= 2; i--)
02110             pc[i] = pc[i - 1] + fnq * pc[i];
02111         pc[1] *= fnq;
02112 /*
02113    Store coefficients in elco and tesco.
02114 */
02115         for (i = 1; i <= nqp1; i++)
02116             elco[nq][i] = pc[i] / pc[2];
02117         elco[nq][2] = 1.;
02118         tesco[nq][1] = rq1fac;
02119         tesco[nq][2] = ((double) nqp1) / elco[nq][1];
02120         tesco[nq][3] = ((double) (nq + 2)) / elco[nq][1];
02121         rq1fac /= fnq;
02122     }
02123     return;
02124 
02125 }               /* end cfode   */

02166 static void prja(int neq, double *y, _lsoda_f f, void *_data)
02167 {
02168     int             i, ier, j;
02169     double          fac, hl0, r, r0, yj;
02170 /*
02171    prja is called by stoda to compute and process the matrix
02172    P = I - h * el[1] * J, where J is an approximation to the Jacobian.
02173    Here J is computed by finite differencing.
02174    J, scaled by -h * el[1], is stored in wm.  Then the norm of J ( the
02175    matrix norm consistent with the weighted max-norm on vectors given
02176    by vmnorm ) is computed, and J is overwritten by P.  P is then
02177    subjected to LU decomposition in preparation for later solution
02178    of linear systems with p as coefficient matrix.  This is done
02179    by dgefa if miter = 2, and by dgbfa if miter = 5.
02180 */
02181     nje++;
02182     ierpj = 0;
02183     jcur = 1;
02184     hl0 = h * el0;
02185 /*
02186    If miter = 2, make n calls to f to approximate J.
02187 */
02188     if (miter != 2) {
02189         fprintf(stderr, "[prja] miter != 2\n");
02190         return;
02191     }
02192     if (miter == 2) {
02193         fac = vmnorm(n, savf, ewt);
02194         r0 = 1000. * fabs(h) * ETA * ((double) n) * fac;
02195         if (r0 == 0.)
02196             r0 = 1.;
02197         for (j = 1; j <= n; j++) {
02198             yj = y[j];
02199             r = max(sqrteta * fabs(yj), r0 / ewt[j]);
02200             y[j] += r;
02201             fac = -hl0 / r;
02202             (*f) (tn, y + 1, acor + 1, _data);
02203             for (i = 1; i <= n; i++)
02204                 wm[i][j] = (acor[i] - savf[i]) * fac;
02205             y[j] = yj;
02206         }
02207         nfe += n;
02208 /*
02209    Compute norm of Jacobian.
02210 */
02211         pdnorm = fnorm(n, wm, ewt) / fabs(hl0);
02212 /*
02213    Add identity matrix.
02214 */
02215         for (i = 1; i <= n; i++)
02216             wm[i][i] += 1.;
02217 /*
02218    Do LU decomposition on P.
02219 */
02220         dgefa(wm, n, ipvt, &ier);
02221         if (ier != 0)
02222             ierpj = 1;
02223         return;
02224     }
02225 }               /* end prja   */

02253    on vectors, with weights stored in the array w.

02274 static void correction(int neq, double *y, _lsoda_f f, int *corflag, double pnorm, double *del, double *delp, double *told,
02275                        int *ncf, double *rh, int *m, void *_data)
02276 /*
02277    *corflag = 0 : corrector converged,
02278               1 : step size to be reduced, redo prediction,
02279               2 : corrector cannot converge, failure flag.
02280 */
02281 
02282 {
02283     int             i;
02284     double          rm, rate, dcon;
02285 
02286 /*
02287    Up to maxcor corrector iterations are taken.  A convergence test is
02288    made on the r.m.s. norm of each correction, weighted by the error
02289    weight vector ewt.  The sum of the corrections is accumulated in the
02290    vector acor[i].  The yh array is not altered in the corrector loop.
02291 */
02292 
02293     *m = 0;
02294     *corflag = 0;
02295     rate = 0.;
02296     *del = 0.;
02297     yp1 = yh[1];
02298     for (i = 1; i <= n; i++)
02299         y[i] = yp1[i];
02300     (*f) (tn, y + 1, savf + 1, _data);
02301     nfe++;
02302 /*
02303    If indicated, the matrix P = I - h * el[1] * J is reevaluated and
02304    preprocessed before starting the corrector iteration.  ipup is set
02305    to 0 as an indicator that this has been done.
02306 */
02307     while (1) {
02308         if (*m == 0) {
02309             if (ipup > 0) {
02310                 prja(neq, y, f, _data);
02311                 ipup = 0;
02312                 rc = 1.;
02313                 nslp = nst;
02314                 crate = 0.7;
02315                 if (ierpj != 0) {
02316                     corfailure(told, rh, ncf, corflag);
02317                     return;
02318                 }
02319             }
02320             for (i = 1; i <= n; i++)
02321                 acor[i] = 0.;
02322         }       /* end if ( *m == 0 )   */
02323         if (miter == 0) {
02324 /*
02325    In case of functional iteration, update y directly from
02326    the result of the last function evaluation.
02327 */
02328             yp1 = yh[2];
02329             for (i = 1; i <= n; i++) {
02330                 savf[i] = h * savf[i] - yp1[i];
02331                 y[i] = savf[i] - acor[i];
02332             }
02333             *del = vmnorm(n, y, ewt);
02334             yp1 = yh[1];
02335             for (i = 1; i <= n; i++) {
02336                 y[i] = yp1[i] + el[1] * savf[i];
02337                 acor[i] = savf[i];
02338             }
02339         }
02340         /* end functional iteration   */
02341         /*
02342            In the case of the chord method, compute the corrector error,
02343            and solve the linear system with that as right-hand side and
02344            P as coefficient matrix.
02345          */ 
02346         else {
02347             yp1 = yh[2];
02348             for (i = 1; i <= n; i++)
02349                 y[i] = h * savf[i] - (yp1[i] + acor[i]);
02350             solsy(y);
02351             *del = vmnorm(n, y, ewt);
02352             yp1 = yh[1];
02353             for (i = 1; i <= n; i++) {
02354                 acor[i] += y[i];
02355                 y[i] = yp1[i] + el[1] * acor[i];
02356             }
02357         }       /* end chord method   */
02358 /*
02359    Test for convergence.  If *m > 0, an estimate of the convergence
02360    rate constant is stored in crate, and this is used in the test.
02361 
02362    We first check for a change of iterates that is the size of
02363    roundoff error.  If this occurs, the iteration has converged, and a
02364    new rate estimate is not formed.
02365    In all other cases, force at least two iterations to estimate a
02366    local Lipschitz constant estimate for Adams method.
02367    On convergence, form pdest = local maximum Lipschitz constant
02368    estimate.  pdlast is the most recent nonzero estimate.
02369 */
02370         if (*del <= 100. * pnorm * ETA)
02371             break;
02372         if (*m != 0 || meth != 1) {
02373             if (*m != 0) {
02374                 rm = 1024.0;
02375                 if (*del <= (1024. * *delp))
02376                     rm = *del / *delp;
02377                 rate = max(rate, rm);
02378                 crate = max(0.2 * crate, rm);
02379             }
02380             dcon = *del * min(1., 1.5 * crate) / (tesco[nq][2] * conit);
02381             if (dcon <= 1.) {
02382                 pdest = max(pdest, rate / fabs(h * el[1]));
02383                 if (pdest != 0.)
02384                     pdlast = pdest;
02385                 break;
02386             }
02387         }
02388 /*
02389    The corrector iteration failed to converge.
02390    If miter != 0 and the Jacobian is out of date, prja is called for
02391    the next try.   Otherwise the yh array is retracted to its values
02392    before prediction, and h is reduced, if possible.  If h cannot be
02393    reduced or mxncf failures have occured, exit with corflag = 2.
02394 */
02395         (*m)++;
02396         if (*m == maxcor || (*m >= 2 && *del > 2. * *delp)) {
02397             if (miter == 0 || jcur == 1) {
02398                 corfailure(told, rh, ncf, corflag);
02399                 return;
02400             }
02401             ipup = miter;
02402 /*
02403    Restart corrector if Jacobian is recomputed.
02404 */
02405             *m = 0;
02406             rate = 0.;
02407             *del = 0.;
02408             yp1 = yh[1];
02409             for (i = 1; i <= n; i++)
02410                 y[i] = yp1[i];
02411             (*f) (tn, y + 1, savf + 1, _data);
02412             nfe++;
02413         }
02414 /*
02415    Iterate corrector.
02416 */
02417         else {
02418             *delp = *del;
02419             (*f) (tn, y + 1, savf + 1, _data);
02420             nfe++;
02421         }
02422     }           /* end while   */
02423 }               /* end correction   */

02473 static void methodswitch(double dsm, double pnorm, double *pdh, double *rh)
02474 {
02475     int             lm1, lm1p1, lm2, lm2p1, nqm1, nqm2;
02476     double          rh1, rh2, rh1it, exm2, dm2, exm1, dm1, alpha, exsm;

02478 /*
