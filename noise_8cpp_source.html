224 void gsl_ran_gaussian_ziggurat_fill (vector_rng& vrng, double sigma, double* restr invals, size_t len)
225 {
226  ALIGNED(invals);
227  for(size_t qq = 0; qq < len; qq++){
228  unsigned long U, sign, i, j;
229  double x, y;
230  while (1) {
231  U = vrng.uint_32();
232  i = U & 0x0000007F; /* 7 bit to choose the step */
233  sign = U & 0x00000080; /* 1 bit for the sign */
234  j = U>>8; /* 24 bit for the x-value */
235 
236  x = j*wtab[i];
237  if (j < ktab[i]){
238  break;
239  }
240 
241  if (PREDICT(i<127, 1)) {
242  double y0, y1;
243  y0 = ytab[i];
244  y1 = ytab[i+1];
245  y = y1+(y0-y1)*vrng.real_64();
246  } else {
247  x = PARAM_R - std::log(1.0-vrng.real_64())/PARAM_R;
248  y = std::exp(-PARAM_R*(x-0.5*PARAM_R))*vrng.real_64();
249  }
250  if (y < std::exp(-0.5*x*x)) {
251  break;
252  }
253  }
254  invals[qq] = sign ? sigma*x : -sigma*x;
255  }
