237 static inline double ran_gauss (vector_rng& vrng, double sigma){
238  uint32_t r32, step, xval;
239  int32_t sign;
240  double x, y;
241  while (1) {
242  r32 = vrng.uint_32();
243  step = r32 & 0x0000007F;
244  sign = r32 & 0x00000080 ? 1 : -1;
245  xval = r32>>8;
246 
247  x = xval*wtab[step];
248  if (xval < ktab[step]){
249  return sign * sigma * x;
250  }
251 
252  if (step<127) {
253  double y0, y1;
254  y0 = ytab[step];
255  y1 = ytab[step+1];
256  y = y1+(y0-y1)*vrng.real_64();
257  } else {
258  x = right_step - std::log(1.0-vrng.real_64())/right_step;
259  y = std::exp(-right_step*(x-0.5*right_step))*vrng.real_64();
260  }
261  if (y < std::exp(-0.5*x*x)) {
262  return sign * sigma * x;
263  }
264  }
265  return sign * sigma*x;
266 }
